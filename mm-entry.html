--- permalink: mmentry hidden: true --- #!/bin/bash UBUNTU_VERSION="22" while getopts v: flag do case "${flag}" in v) UBUNTU_VERSION=${OPTARG};; esac done if [ $UBUNTU_VERSION -eq "22" ]; then curl https://code.webb.tj/mm-ubuntu-22 > /usr/local/bin/webbtj-init elif [ $UBUNTU_VERSION -eq "20" ]; then curl https://code.webb.tj/mm-ubuntu-22 > /usr/local/bin/webbtj-init elif [ $UBUNTU_VERSION -eq "18" ]; then curl https://code.webb.tj/mm-ubuntu-22 > /usr/local/bin/webbtj-init else echo "\033[0;41mPlease only pass a valid Ubuntu version (22, 20, or 18) or omit the -v flag for the default install (22).\033[m" exit fi chmod +x /usr/local/bin/webbtj-init echo "\033[0;42mInitialization tool installed.\033[m" echo "Please run \033[0;33mwebbtj-init\033[m"
